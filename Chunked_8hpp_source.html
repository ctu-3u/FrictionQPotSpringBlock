465 template <class T>
466 void rshift_y(long istart, const T& y, size_t nbuffer = 0)
467 {
468 QPOT_ASSERT(y.size() > 0);
469 QPOT_ASSERT(istart <= m_istop);
470 QPOT_ASSERT(istart >= m_istart); // otherwise the function does not what it's name suggests
471
472// if there is overlap between the chunks of Y in memory and input:
473// check that the values are indeed the same
474#ifdef QPOT_ENABLE_DEBUG
475 if (istart < m_istop) {
476 size_t d = static_cast<size_t>(m_istop - istart);
477 xt::xtensor<double, 1> o = xt::empty<double>({d});
478 xt::xtensor<double, 1> n = xt::empty<double>({d});
479 std::copy(m_y.cend() - d, m_y.cend(), o.begin());
480 std::copy(y.cbegin(), y.cbegin() + d, n.begin());
481 QPOT_DEBUG(xt::allclose(o, n));
482 }
483#endif
484
485 // supplied chunk smaller than the chunk in memory
486 if (istart >= m_istart && istart + static_cast<long>(y.size()) <= m_istop) {
487 return;
488 }
489
490 m_istart = m_istop - static_cast<long>(nbuffer);
491
492 // take everything for "y": it spans the new entries and the buffered old entries
493 if (istart <= m_istart) {
494 return this->set_y(istart, y);
495 }
496
497 // buffer "m_y[m_n - nbuffer:]" and take the rest of "y"
498 m_ymin_data = *y.cbegin();
499 QPOT_ASSERT(nbuffer <= m_n);
500 QPOT_DEBUG(std::is_sorted(y.cbegin(), y.cend()));
501 size_t d = static_cast<size_t>(m_istop - istart); // items to skip from "y"
502 m_istop = istart + static_cast<long>(y.size());
503 m_n = static_cast<size_t>(m_istop - m_istart);
504 QPOT_ASSERT(m_n >= nbuffer);
505
506 std::copy(m_y.cend() - nbuffer, m_y.cend(), m_y.begin());
507 m_y.resize(m_n);
508 std::copy(y.cbegin() + d, y.cend(), m_y.begin() + nbuffer);
509
510 m_full_search = true;
511 this->update();

979private:
980 void update()
981 {
982 if (m_x <= m_y.front()) {
983 m_redraw = -1;
984 }
985 else if (m_x > m_y.back()) {
986 m_redraw = +1;
987 }
988 else {
989 m_redraw = 0;
990 }
991
992 m_lock = m_redraw != 0;
993
994 if (m_lock) {
995 return;
996 }
997
998 if (m_full_search) {
999 m_li = std::lower_bound(m_y.cbegin(), m_y.cend(), m_x) - m_y.cbegin() - 1;
1000 m_left = m_y[m_li];
1001 m_right = m_y[m_li + 1];
1002 m_full_search = false;
1003 return;
1004 }
1005
1006 if (m_left < m_x && m_right >= m_x) {
1007 return;
1008 }
1009
1010 size_t l = m_li > m_proximity ? m_li - m_proximity : 0;
1011 size_t r = std::min(m_li + m_proximity, m_n - 1);
1012
1013 if (m_y[l] < m_x && m_y[r] >= m_x) {
1014 m_li = std::lower_bound(&m_y[l], &m_y[l] + r - l, m_x) - &m_y[l] - 1 + l;
1015 }
1016 else {
1017 m_li = std::lower_bound(m_y.cbegin(), m_y.cend(), m_x) - m_y.cbegin() - 1;
1018 }
1019
1020 m_left = m_y[m_li];
1021 m_right = m_y[m_li + 1];
1022 }
